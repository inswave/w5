Defaults

defaults 를 이용하여 Row 생성 시 값을 지정하지 않은 컬럼에 default value 를 설정한다.
다음과 같이 3가지 방식을 사용할 수 있다.

Column Model

width, headerLabel 같이 그리드 생성 시점에 설정하는 colModel attribute 에 해당 컬럼에 사용될 default value 를 지정한다.

// firstName, lastName, companyName 에 default 를 지정한다.
var colModel = [{ "width": 80, "headerLabel": "firstName", "default": "shye", "style": "font-weight: bold" },
                { "width": 80, "headerLabel": "lastName", "default": "lee" },
                { "width": 140, "headerLabel": "companyName", "default": "inswave" },
                { "width": 150, "headerLabel": "address" },
                ....];

// colModel 지정은 일반적인 방법과 동일하다.              
var grid1 = new w5.Grid({
    ....
    colModel : colModel
}).render();

// 입력하고자 하는 데이타를 준비한다.
// firstName, lastName, companyName 에 대한 값이 존재하지 않는다.
var tempData = { "address": "6 Greenleaf Ave", "city": "San Jose", "county": "Santa Clara", "state": "CA", "zip": "95111", "phone1": "408-540-1785", "phone2": "408-813-4592", "email": "vinouye@aol.com", "web": "http://www.cnetworkinc.com" };

// 2번째 위치에 Row 를 삽입한다.
// 누락된 컬럼의 데이타는 기본값으로 setting 된다.
grid1.addRow( 1, tempData );

setDefaults / getDefaults

setDefaults() 메소드를 이용하여 동적으로 defaults value 를 지정한다.
그리드 생성 후 지정도 가능하다.
Object 는 reference 로 전달됨으로 모든 인스턴스(Model)가 공유한다. 따라서 Backbone 에서는 function 으로 defaults 를 정의하는 것을 권장한다.

    defaults = function () {
      return {
        firstName: 'youngsoo',
        lastName: 'choi',
        companyName: 'inswave systems'
      };
    };

    // defaults 로 함수를 전달한다.
    // 함수가 return 하는 Object 의 기본 값이 이후 Row add 시 적용된다.
    grid1.setDefaults( defaults );

Backbone.Model

Backbone.Model 을 이용하여 defaults 를 설정한다.
위의 Column Model 이나 setDefaults 방식보다 우선순위가 높아서 여러 방식으로 defaults 가 설정된 경우, Model 에 extend 로 설정한 내용이 적용된다.

    // Model extend 에 defaults 를 설정한다.
    model = Backbone.Model.extend( { defaults: function() {
      return {
        firstName: 'A',
        lastName: 'B',
        companyName: 'C'
      };
    } } ),

    // Collection 생성 시점에 위의 모델을 지정한다.
    collection = new Backbone.Collection( JSON.parse(data), { model: model } ),
